HTML


    <div class="tabs">
        <ul>
            <li>Tab1</li>
            <li>Tab2</li>
            <li>Tab3</li>
        </ul>
        <div>
            <div>Content1</div>
            <div>Content2</div>
            <div>Content3</div>
        </div>
    </div>
    <button id="b1">select Tab1</button>
    <button id="b2">select Tab2</button>
    <button id="b3">select Tab3</button>
    <button id="b4">get selected</button>
    <div class="log"></div>

JavaScript


    var tabs = $('.tabs');
    tabs.bizTab({
        theme: 'brown'
    }).on('change', function(e, data) {
        $('.log').html('index: ' + data.index + ', title: ' + data.title + ', content: ' + data.content);
    });
    $('button').bizButton({
        theme: 'blue-gray'
    });
    $('#b1').click(function() {
        tabs.bizTab('index', 0);
    });
    $('#b2').click(function() {
        tabs.bizTab('index', 1);
    });
    $('#b3').click(function() {
        tabs.bizTab('index', 2);
    });
    $('#b4').click(function() {
        var data = tabs.bizTab('index');
        bizui.alert('index: ' + data.index + ', title: ' + data.title + ', content: ' + data.content);
    });

Options

参数名称 参数类型 描述
[action] String 切换方式（click | hover），默认 'click'
[customClass] String 自定义 class
[selectedIndex] Number 选中 tab 序号，默认 0
[theme] String 主题

Methods

destroy()

销毁

index([Number selectedIndex][, Boolean fire])

获取/设置当前被选中的 tab


Events

change(e, data)
